Query1 = '''SELECT
                l.zip_code,
                ROUND(AVG(s.monthly_charge),3) as avgMonth
            FROM
                services s
            JOIN
                location l ON s.customer_id = l.customer_id
            WHERE
                s.tenure_in_months >= 12
                AND s.online_security = 'Yes'
                AND s.online_backup = 'Yes'
            GROUP BY
                l.zip_code
            ORDER BY
                avgMonth DESC
            LIMIT 5;'''
